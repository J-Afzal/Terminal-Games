194 void TicTacToe::ExecuteGeneralCommand(const std::tuple<uint32_t, uint32_t> &p_command)
195 {
196 const auto COMMAND_FIND_LOCATION = Globals::ImplementStdRangesFind(m_commandsRemaining.begin(), m_commandsRemaining.end(), p_command);
197 const uint32_t ROW = std::get<0>(p_command);
198 const uint32_t COLUMN = std::get<1>(p_command);
199
200 if (m_currentPlayer == Globals::G_TICTACTOE_PLAYER_X)

202 m_gameGrid.at(ROW).at(COLUMN) = Globals::G_TICTACTOE_GRID_PLAYER_X_OCCUPIED;
